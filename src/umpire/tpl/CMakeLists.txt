set (json_headers
  ./json/json.hpp )

blt_add_library(
  NAME umpire_tpl_json
  HEADERS ${json_headers})

target_link_libraries(umpire_tpl_json INTERFACE)

  ${json_headers}
  DESTINATION include/umpire/tpl/json)

  umpire_tpl_json
